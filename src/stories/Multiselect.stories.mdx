### Multiselect: colors

<Preview>
  <Story name="colors">
    {() => {
      const options = [
        { value: 'chocolate', 
          label: 'Chocolate'
        },
        { value: 'strawberry', 
          label: 'Strawberry',
          color: `${theme.colors.notification}`
           },
        { value: 'vanilla', 
          label: 'Vanilla',
          color: `${theme.colors.danger}`
        }
      ];
      const colorStyles = {
        multiValue: (styles, { data }) => {
        const color = data.color;
        return {
          ...styles,
          backgroundColor: color ? color : theme.colors.primary,
        };
      },
      };
      return (
        <div style={{height: '150px'}}>
          <Select 
            isMulti 
            defaultValue={[options[1], options[2]]}
            options={options}
            styles={colorStyles}
          >
          </Select>
        </div>
      );
    }}
  </Story>
</Preview>
