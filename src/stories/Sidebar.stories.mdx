## Props

```typescript
interface SidebarProps {
  children?: React.ReactNode;
  title?: string;
  isOpen?: boolean;
  side?: "right" | "left";
  style: ReactModal.Styles;
}

interface SidebarButtonProps {
  position?: "topLeft" | "topRight" | "bottomLeft" | "bottomRight";
  onClick: (e: any) => void;
}
```
