### Multiselect: colors

https://github.com/JedWatson/react-select/blob/master/docs/examples/StyledMulti.tsx

In TypeScript, options need to be properly typed:

```tsx
const colorOptions: SelectColoredOption[] = [
  ...
];
```

Custom styles can also be specified:

```tsx
const colorStyles = {
  multiValue: (styles, { data }) => {
    const color = data.color;
    return {
      ...styles,
      backgroundColor: color ? color : theme.color.interactive.primary,
    };
  },
};

return (
  <Select
    isMulti
    defaultValue={[colorOptions[1], colorOptions[2]]}
    options={colorOptions}
    styles={colorStyles}
  />
);
```

## Using ref

In TypeScript, Ref needs to be properly typed:

```tsx
const selectRef = React.useRef<SelectInstance<typeof options[number]> | null>(
  null
);
```

<Canvas>
  <Story name="ref">
    {() => {
      const options = [
        { value: "chocolate", label: "Chocolate" },
        { value: "strawberry", label: "Strawberry" },
        { value: "vanilla", label: "Vanilla" },
      ];
      const selectRef = React.useRef(null);
      return (
        <div style={{ height: "150px" }}>
          <Select
            isMulti
            defaultValue={[options[1], options[2]]}
            options={options}
            ref={selectRef}
          ></Select>

        </div>
      );
    }}
