### Multiselect: colors

https://github.com/JedWatson/react-select/blob/master/docs/examples/StyledMulti.tsx

In TypeScript, options need to be properly typed:

```tsx
const colorOptions: SelectColoredOption[] = [
  ...
];
```

Custom styles can also be specified:

```tsx
const colorStyles = {
  multiValue: (styles, { data }) => {
    const color = data.color;
    return {
      ...styles,
      backgroundColor: color ? color : theme.color.interactive.primary,
    };
  },
};

return (
  <Select
    isMulti
    defaultValue={[colorOptions[1], colorOptions[2]]}
    options={colorOptions}
    styles={colorStyles}
  />
);
```
