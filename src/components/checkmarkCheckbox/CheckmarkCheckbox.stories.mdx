# CheckmarkCheckbox

### CheckmarkCheckbox properties

|   properties | required | type    | description                                                 |
| -----------: | :------: | :------ | :---------------------------------------------------------- |
|    labelText |  false   | string  | Checkbox label                                              |
| withoutLabel |  false   | boolean | Checkbox without label with larger clickable surface        |
|       double |  false   | boolean | If set to true will show double checkmark instead of single |

# CheckboxGroup

### CheckboxGroup properties

| properties | required | type                | description |
| ---------: | :------: | :------------------ | :---------- |
|   children |  false   | `ReactElement<any>` |             |

<Canvas>
  <Story name="group">
    {() => {
      const [value, setValue] = React.useState({
        0: false,
        1: false,
        2: false,
      });
      const handleCheckbox = (index) =>
        setValue({ ...value, [index]: !value[index] });
      const labels = ["New York", "San Francisco", "Los Angeles"];
      const names = ["checkbox-1", "checkbox-2", "checkbox-3"];
      return (
        <FlexContainer
          just
          style={{
            padding: "3rem",
          }}
        >
          <CheckboxGroup name="gnui-checkbox-group">
            {names.map((names, index) => (
              <CheckmarkCheckbox
                id={names[index]}
                onChange={() => handleCheckbox(index)}
                labelText={labels[index]}
              />
            ))}
          </CheckboxGroup>
        </FlexContainer>
      );
    }}
  </Story>
</Canvas>
