### Multiselect

<Canvas>
  <Story name="multiselect">
    {() => {
      const options = [
        { value: "chocolate", label: "Chocolate" },
        { value: "strawberry", label: "Strawberry" },
        { value: "vanilla", label: "Vanilla" },
      ];
      return (
        <div style={{ height: "150px" }}>
          <Select
            isMulti
            defaultValue={[options[1], options[2]]}
            options={options}
          ></Select>
        </div>
      );
    }}
  </Story>
</Canvas>

### Multiselect: colors

<Canvas>
  <Story name="colors">
    {() => {
      const options = [
        { value: "chocolate", label: "Chocolate" },
        {
          value: "strawberry",
          label: "Strawberry",
          color: `${theme.color.interactive.info}`,
        },
        {
          value: "vanilla",
          label: "Vanilla",
          color: `${theme.color.interactive.error}`,
        },
      ];
      const colorStyles = {
        multiValue: (styles, { data }) => {
          const color = data.color;
          return {
            ...styles,
            backgroundColor: color ? color : theme.color.interactive.primary,
          };
        },
      };
      return (
        <div style={{ height: "150px" }}>
          <Select
            isMulti
            defaultValue={[options[1], options[2]]}
            options={options}
            styles={colorStyles}
          ></Select>
        </div>
      );
    }}
  </Story>
</Canvas>
